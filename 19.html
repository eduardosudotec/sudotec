//swith e uma estrutura de controle de fluxo //quepermite que oprograma execute diferentes acoes //com base em uma determinada codicao. //ele verifica uma condicao e , depende do resultado //executa um determinado bloco de codigo. int dia; printf("digite um numero de 1 a7 :"); scanf("%d" , &dia ); switch(dia); printf("doomingo"); break; case 2 : printf("segunda"); break; case 3 : printf("terca"); break; case 4 : printf("quarta"); break; case 5 : printf("quinta"); break; case 6 : printf("sexta"); break; case 7 : printf("sabado"); break; default: printf("entrada invalida") }
